CREATE TABLE GOODS (
	ID INT PRIMARY KEY,
	"NAME" TEXT
);

CREATE TABLE TYPES (
	ID INT PRIMARY KEY,
	"NAME" TEXT
);

INSERT INTO TYPES

	(1, 'Язык программирования'),
	(2, 'Самолет'),
	(3, 'Видеокарта');

CREATE TABLE GOODSTYPES (
	GOODSID INT FOREIGN KEY REFERENCES GOODS(ID),
	TYPESID INT FOREIGN KEY REFERENCES TYPES(ID),
	PRIMARY KEY (GOODSID, TYPESID)
);

INSERT INTO GOODSTYPES
VALUES
	(1, 1),
	(1, 3),
	(2, 2);

SELECT G."NAME", T."NAME"
FROM GOODS G
LEFT JOIN GOODSTYPES GT
	ON G.ID = GT.GOODSID
LEFT JOIN TYPES T
	ON GT.TYPESID = T.ID;
